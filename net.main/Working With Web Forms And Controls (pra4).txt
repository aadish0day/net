Q1] VALIDATION CONTROLS
Creating design and adding validators:
Add ControlToValidate to all (name of input box)
For empty fields use RequiredFieldValidator
For matching specific values use RegularExpressionValidator (change validationExpression based on needs eg- mobile, email)
For age range validation use RangeValidator (add min, max value)
For confirming password use CompareVlidator (controlToCompare - password and ControlToValidate - confirmPassword)

    <form id="form1" runat="server">
    <div>

        <h1>Calendar Control:</h1>

        <asp:Calendar ID="Calendar1" runat="server" 
            OnDayRender="Calendar1_DayRender">
        </asp:Calendar>

        <br />

        <asp:Button ID="Button1" runat="server" Text="Show Selected Date" 
            OnClick="Button1_Click" /><br /><br />

        <asp:Label ID="Label1" runat="server" Text=""></asp:Label><br />
        <asp:Label ID="Label2" runat="server" Text=""></asp:Label>

    </div>
    </form>
</body>
</html>


# Default.aspx.cs:

using System;
using System.Web.UI;
using System.Web.UI.WebControls;

public partial class _Default : System.Web.UI.Page
{
    protected void Page_Load(object sender, EventArgs e)
    {
        // This label shows current date
        Label2.Text = "Today's Date: " + DateTime.Now.ToString("dd-MM-yyyy");
    }

    protected void Button1_Click(object sender, EventArgs e)
    {
        // Read selected date from Calendar
        DateTime selectedDate = Calendar1.SelectedDate;

        // If no date is selected, show message
        if (selectedDate == DateTime.MinValue)
        {
            Label1.Text = "No date selected!";
        }
        else
        {
            Label1.Text = "Selected Date: " + selectedDate.ToString("dd-MM-yyyy");
        }
    }

    protected void Calendar1_DayRender(object sender, DayRenderEventArgs e)
    {
        // Christmas Day
        if (e.Day.Date.Day == 25 && e.Day.Date.Month == 12)
        {
            e.Cell.Controls.Add(new LiteralControl("<br/> Christmas"));
        }

        // Highlight range: Dec 23-28 (2025)
        if (e.Day.Date > new DateTime(2025, 12, 23) &&
            e.Day.Date <= new DateTime(2025, 12, 28))
        {
            e.Cell.BackColor = System.Drawing.Color.Black;
            e.Cell.ForeColor = System.Drawing.Color.Green;
            e.Cell.BorderWidth = 2;
        }
    }
}
