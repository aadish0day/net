 Step 2: Add AccountController
# AccountController.cs:

using System;
using System.Web;
using System.Web.Mvc;

namespace WebApplication2.Controllers
{
    public class AccountController : Controller
    {
        [HttpGet]
        public ActionResult Login()
        {
            [cite_start]// Check if the 'usernameCookie' already exists on the user's browser 
            if (Request.Cookies["usernameCookie"] != null)
            {
                [cite_start]// Retrieve the cookie value and store it in ViewBag to pre-fill the form 
                ViewBag.CookieUser = Request.Cookies["usernameCookie"].Value;
            }

            return View();
        }

        [HttpPost]
        public ActionResult Login(string username, string remember)
        {
            [cite_start]// Store the username in a Session variable for temporary state management 
            Session["username"] = username;

            [cite_start]// If the "remember" checkbox was checked (set to "on") 
            if (remember == "on")
            {
                [cite_start]// Create a new persistent cookie named 'usernameCookie' 
                HttpCookie cookie = new HttpCookie("usernameCookie");
                cookie.Value = username;
                
                [cite_start]// Set the cookie to expire 7 days from now 
                cookie.Expires = DateTime.Now.AddDays(7);
                
                [cite_start]// Add the cookie to the response to save it on the user's machine 
                Response.Cookies.Add(cookie);
            }

            [cite_start]// Redirect the user to the Welcome action 
            return RedirectToAction("Welcome");
        }

        public ActionResult Welcome()
        {
            [cite_start]// Retrieve the username from the Session to display on the Welcome page 
            ViewBag.User = Session["username"];
            return View();
        }

        public ActionResult Logout()
        {
            [cite_start]// Clear all data stored in the current Session 
            Session.Clear();
            
            [cite_start]// Redirect the user back to the Login page 
            return RedirectToAction("Login");
        }
    }
}
