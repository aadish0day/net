Step 2: In the model folder add a class named as GuestResponse.cs

# GuestResponse.cs:

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace WebApplication7.Models
{
    public class GuestResponse
    {
        public string Name { get; set; }
        
        public string Email { get; set; }
        
        public string Phone { get; set; }
        
        public bool? WillAttend { get; set; }
    }
}

Make A Controller 
Make an Index View In Views Folder (We can also directly create index view by right clicking on index method in controller)

# HomeController.cs:

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;
using WebApplication7.Models; // Essential: This links your model to the controller

namespace WebApplication7.Controllers
{
    public class HomeController : Controller
    {
        // GET: Home/Index
        public ActionResult Index()
        {
            return View();
        }

        // GET: Home/RsvpForm
        [HttpGet]
        public ActionResult RsvpForm()
        {
            return View();
        }

        // POST: Home/RsvpForm
        [HttpPost]
        public ActionResult RsvpForm(GuestResponse guestResponse)
        {
            // Check if the form data is valid (e.g., email is correctly formatted)
            if (ModelState.IsValid)
            {
                // If valid, send the data to a "Thanks" page
                return View("Thanks", guestResponse);
            }
            else
            {
                // If there are errors, stay on the form page
                return View();
            }
        }
    }
}




# Right Click on RscpForm in HttpGet Method in controller and make a view for that method with below specifications:

# Views/Home/RsvpForm.cshtml file:

@model WebApplication7.Models.GuestResponse

@{
    Layout = null;
}

<!DOCTYPE html>
<html>
<head>
    <meta name="viewport" content="width=device-width" />
    <title>RSVP Form</title>
</head>
<body>
    <div> 
        @using (Html.BeginForm()) 
        {
            <h3>RSVP Now!</h3>
            
            <p>Your Name: @Html.TextBoxFor(x => x.Name)</p>
            <p>Your Email: @Html.TextBoxFor(x => x.Email)</p>
            <p>Your Phone: @Html.TextBoxFor(x => x.Phone)</p>
            
            <p>

                }, "Choose an option")
            </p>
            
            <input type="submit" value="Submit RSVP" />
        }
    </div>
</body>
</html>
